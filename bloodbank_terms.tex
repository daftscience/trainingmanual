
\newcommand{\bbappitem}[1]{\item \appicon{bb/#1}  {{\Large{\boldcap{\gls{#1}}}}\\}\glsdesc{#1} }


\newglossaryentry{bb_label_verify}
{
    name={Label Verify},
    description={A Blood Bank application that verifies the label affixed to blood bag after modification or pooling. It does this by comparing the details on the label with the values stored in the system.}
}

\newglossaryentry{bb_modify_crossmatch}{
    name={Modify Crossmatch},
    description={A Blood Bank application that reinstates a completed crossmatch and updates the crossmatch expiration date and time for a product. }
}
\newglossaryentry{bb_modify_products}{
    name={Modify Products},
    description={A Blood Bank application to modify or pool products. New products can be created from existing products in inventory, split products, add product attributes, convert autologous and directed products to allogeneic products, and create single pooled products from multiple component products. }
}

\newglossaryentry{bb_order_inquiry}
{
  name={Order Inquiry},
  description={A Blood Bank application to display detailed information about product orders. Order Inquiry reports can be generated to print details about the orders displayed.}
}
\newglossaryentry{bb_product_orders}
{
  name={Product Orders},
  description={A Blood Bank application to place on-demand orders for laboratory and blood bank procedures directly on blood bank products, without an association to a specific patient.}
}

\newglossaryentry{bb_patient_product_inquiry}
{
  name={Patient Product Inquiry},
  description={A Blood Bank application to view a list of products currently associated with a patient. Antibodies, Blood Bank comments, and transfusion requirements can also be viewed.}
}

\newglossaryentry{bb_product_history_review}
{
  name={Product History Review},
  description={A Blood Bank application used to display a detailed overview of current product activity.}
}
\newglossaryentry{bb_quality_control_result_entry}
{
  name={Quality Control Result Entry},
  description={A Blood Bank application to create and maintain quality control (QC) groups and to enter results for selected QC groups.}
}

\newglossaryentry{bb_qc_inquiry}
{
  name={QC Inquiry},
  description={A Blood Bank application used by supervisory personnel to review Blood Bank quality control results entered in Quality Control Result Entry.}
}

\newglossaryentry{bb_quarantine_products}
{
  name={Quarantine Products},
  description={A Blood Bank application used to add or remove the ``Quarantined'' state on blood bank products.}
}

\newglossaryentry{bb_receive_products}{
    name={Receive Products},
    description={A Blood Bank application used to receive products into the Blood Bank inventory. It can also print a report of products received as well as component tags and pilot labels.}
}

\newglossaryentry{bb_correct_inventory}{
    name={Correct Inventory},
    description={A Blood Bank application to modify existing product information. This application can unlock products, change product demographic information, change final disposition information, update emergency dispense information, update special testing information and add to pooled product.}
}


\newglossaryentry{bb_dispense_and_assign_products}{
    name={Dispense and Assign Products},
    description={A Blood Bank application to dispense and assign products to patients for future use. It can be used to dispense products to patients in an emergency situation.}
}

\newglossaryentry{bb_final_disposition}
{
    name={Final Disposition},
    description={A Blood Bank application used to perform three different updates to blood products: \textbf{Transfuse, Dispose and Destroy}.}
}


\newglossaryentry{bb_generate_tags_and_labels}{
    name={Generate Tags and Labels},
    description={A Blood Bank application to create tags and labels on-demand.}
}

\newglossaryentry{bb_inventory_search}{
    name={Inventory Search},
    description={A Blood Bank application used to search the Blood Bank Inventory for specific products.}
}

\newglossaryentry{bb_release_products}{
    name={Release Products},
    description={A Blood Bank application used to release products from an active Crossmatched or assigned state.}
}

\newglossaryentry{bb_report_selection}{
    name={Report Selection},
    description={A Blood Bank application to print blood bank reports that are used to assist in inventory management and supervisory review of patient-related events.}
}


\newglossaryentry{bb_result_entry}{
    name={Blood Bank Result Entry},
    description={A Blood Bank application used to enter and correct results for blood bank procedures ordered on patients or blood bank products. }
}
